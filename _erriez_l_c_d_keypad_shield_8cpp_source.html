66 LCDButton LCDKeypadShield::getButtons()
67 {
68  static LCDButton keyLast = ButtonNone;
69  static unsigned long keyTimeLast = 0;
70  int analogKey;
71  LCDButton key;
72 
73  analogKey = analogRead(0);
74 
75  if ((millis() - keyTimeLast) < 100) {
76  return keyLast;
77  }
78 
79  keyTimeLast = millis();
80 
81  if (analogKey < 50) {
82  key = ButtonRight;
83  } else if (analogKey < 200) {
84  key = ButtonUp;
85  } else if (analogKey < 300) {
86  key = ButtonDown;
87  } else if (analogKey < 500) {
88  key = ButtonLeft;
89  } else if (analogKey < 700) {
90  key = ButtonSelect;
91  } else {
92  key = ButtonNone;
93  }
94 
95  if (key == keyLast) {
96  return key;
97  } else {
98  keyLast = key;
99  return ButtonNone;
100  }
101 }
